cmake_minimum_required(VERSION 3.9)
project(MSOCKET)
set(CMAKE_EXPORT_COMPILE_COMMANDS true)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/lib)
add_subdirectory(SocketLib)

set(Client_Src SocketClient)
set(Server_Src SocketServer)
add_executable(SocketClient ${Client_Src}/SocketClient.cpp)
target_link_libraries(SocketClient Socket)

add_executable(SocketServer ${Server_Src}/Server.cpp)
target_link_libraries(SocketServer Socket)
